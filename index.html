Q:let studentRecords = [ {name: 'John', id: 123, marks : 98 }, {name: 'Baba', id: 101, marks : 23 }, {name: 'yaga', id: 200, marks : 45 }, {name: 'Wick', id: 115, marks : 75 } ]

Q1:We are interested in retrieving only the students' names; all the names should be in caps. ['JOHN', 'BABA', 'YAGA', 'WICK'] 

    Ans:
    let ans = [];
    for(let i=0;i < studentRecords.length;i++){
        ans.push(studentRecords[i].name.toUpperCase());
    }
    console.log(ans);
